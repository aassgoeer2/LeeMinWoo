#include <stdio.h>

void main()
{
	int a = 0, b = 0, c = 0;
	int Sum, Aver;
	char Grade;

	scanf("%d%d%d", &a, &b, &c);
	Sum = a + b + c;
	Aver = (int)Sum / 3;
	if (Aver > 90)
		Grade = 'A';
	else if (Aver > 80)
		Grade = 'B';
	else if (Aver > 70)
		Grade = 'C';
	else if (Aver > 60)
		Grade = 'D';
	else
		Grade = 'F';

	printf("합계 : %d\n", Sum);
	printf("평균 : %d\n", Aver);
	printf("등급 : %d\n", Grade);
}
